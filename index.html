function doPost(e) { try { // no-cors 模式下資料會放在 postData.contents var raw = ""; if (e.postData && e.postData.contents) { raw = e.postData.contents; } else { raw = JSON.stringify(e.parameter); } var data = JSON.parse(raw); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var timestamp = new Date(); // 整理出工項目與人數 var workerStr = ""; try { var counts = JSON.parse(data.workerCount || "{}"); var types = (data.workType || "").split("、").filter(function(w) { return w !== ""; }); workerStr = types.map(function(w) { return counts[w] ? w + ":" + counts[w] + "人" : w; }).join("、"); } catch(err) { workerStr = data.workType || ""; } var rowData = [ timestamp, data.date || '', data.weather || '', data.project || '', data.workCategory || '', workerStr, data.content || '' ]; sheet.appendRow(rowData); return ContentService .createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doOptions(e) { return ContentService .createTextOutput("") .setMimeType(ContentService.MimeType.TEXT); }
